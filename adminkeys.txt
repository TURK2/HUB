local function LoadKeysFromGitHub()
    local s, data = pcall(function()
        return game:HttpGet(KeyURL)
    end)

    if not s then
        warn("‡πÇ‡∏´‡∏•‡∏î‡∏Ñ‡∏µ‡∏¢‡πå‡∏•‡πâ‡∏°‡πÄ‡∏´‡∏•‡∏ß: ", data)

    for line in string.gmatch(data, "[^\r\n]+") do

    end

    return keys
end
